+        val matrix = imageMatrix
+

+        val viewHeight = (measuredHeight - paddingTop - paddingBottom).toFloat()
+        val drawableWidth = drawable.intrinsicWidth.toFloat()
+        val drawableHeight = drawable.intrinsicHeight.toFloat()

+        val scale = when {
