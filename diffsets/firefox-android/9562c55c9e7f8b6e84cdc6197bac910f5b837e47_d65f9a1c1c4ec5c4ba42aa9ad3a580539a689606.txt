+import android.content.Context
+import android.util.AtomicFile
+import org.json.JSONException
+import org.json.JSONObject
+import java.io.File
+import java.io.FileNotFoundException
+import java.io.IOException
+
+class MobileMetricsPingStorage(
+        private val context: Context,
+        file: File = File("${context.cacheDir}/$STORAGE_FOLDER/$FILE_NAME")
+) {
+    private val atomicFile = AtomicFile(file)
+
+    suspend fun save(json: JSONObject) {
+        val stream = atomicFile.startWrite()
+        try {
+            stream.writer().use {
+                it.append(json.toString())
+            }
+
+            atomicFile.finishWrite(stream)
+        } catch (e: IOException) {
+            atomicFile.failWrite(stream)
+        }
+    }
+
+    suspend fun load(): JSONObject? {
+        return try {
+            JSONObject(String(atomicFile.readFully()))
+        } catch (e: FileNotFoundException) {
+            null
+        } catch (e: JSONException) {
+            null
+        }
+    }
+
+    fun remove() {
+        atomicFile.delete()
+    }
+
+    companion object {
+        val STORAGE_FOLDER = "mobile-metrics"
+        val FILE_NAME = "metrics.json"
+    }
+}
