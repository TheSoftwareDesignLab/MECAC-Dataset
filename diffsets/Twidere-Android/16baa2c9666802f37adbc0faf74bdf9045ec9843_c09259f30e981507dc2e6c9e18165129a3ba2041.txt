+        }

+        public void installViewFactory() {
+            final LayoutInflater inflater = LayoutInflater.from(mContext);
+            if (inflater.getFactory() == null) {
+                LayoutInflaterCompat.setFactory(inflater, new ThemedLayoutInflaterFactory(themed, this));
+            }
+        }
