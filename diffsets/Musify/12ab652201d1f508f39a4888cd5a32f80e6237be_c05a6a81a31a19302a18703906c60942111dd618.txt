+  assert(value >= 0 && value <= 1);

+  final hsl = HSLColor.fromColor(color);
+  final hslDark = hsl.withLightness(
+    (darker ? (hsl.lightness - value) : (hsl.lightness + value))

+  );
+

+    900: getShade(color, value: 0.25, darker: true),
+  };
