+      setColumnWidths((columnWidths) => {

+        }
+
+        const newColumnWidths = [...columnWidths];
+
+        newColumnWidths[index] = width;

+  );
+
+  return {
+    columnWidth,
+    resizeColumnWidth,
+  };
