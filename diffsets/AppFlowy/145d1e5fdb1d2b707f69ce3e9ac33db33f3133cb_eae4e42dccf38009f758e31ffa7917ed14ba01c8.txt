+}

+while [[ $# -gt 0 ]]; do

+        ;;
+    --app-identifier)

+        shift 2
+        ;;
+    --company-name)
+        COMPANY_NAME="$2"
+        shift 2
+        ;;
+    --copyright)
+        COPYRIGHT="$2"
+        shift 2
+        ;;
+    --icon-path)
+        ICON_PATH="$2"
+        shift 2
+        ;;
+    --output-dir)

+        shift 2
+        ;;
+    --help)

+        exit 0
+        ;;
+    *)

+        show_usage
+        exit 1
+        ;;

+    echo -e "\033[31mError: Application identifier is required\033[0m"

+    echo -e "\033[31mError: Company name is required\033[0m"

+if [ -z "$COPYRIGHT" ]; then
+    echo -e "\033[31mError: Copyright information is required\033[0m"

+update_runner_files() {
