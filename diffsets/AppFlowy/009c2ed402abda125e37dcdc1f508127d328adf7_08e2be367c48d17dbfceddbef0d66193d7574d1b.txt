+      }
+      if (mimeType!.contains('text')) {

+        if (mimeType!.contains('pdf') ||
+            mimeType!.contains('doc') ||
+            mimeType!.contains('docx')) {

+        }
+        if (mimeType!.contains('rtf')) {
+          return FileType.text;
+        }
+      }

+      return FileType.other;
+    }
