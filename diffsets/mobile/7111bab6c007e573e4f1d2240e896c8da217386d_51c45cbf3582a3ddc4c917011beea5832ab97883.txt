+class PuzzleSettingsScreen extends StatelessWidget {

+  final UserId userId;
+
+  @override
+  Widget build(BuildContext context) {

+class _Body extends ConsumerWidget {
+  const _Body({required this.userId});
+
+  final UserId userId;
+
+  @override
+  Widget build(BuildContext context, WidgetRef ref) {
+    final autoNext = ref.watch(
+      PuzzlePreferencesProvider(userId).select((value) => value.autoNext),
+    );
+
+    return SafeArea(
+      child: ListView(
+        children: [
+          ListSection(
+            hasLeading: true,
+            showDivider: true,
+            children: [
+              SwitchSettingTile(
+                title: const Text('Jump to next puzzle immediately'),
+                value: autoNext,
+                onChanged: (value) {
+                  ref
+                      .read(PuzzlePreferencesProvider(userId).notifier)
+                      .setAutoNext(value);
+                },
+              ),
+            ],
+          ),
+        ],
+      ),
+    );
+  }
