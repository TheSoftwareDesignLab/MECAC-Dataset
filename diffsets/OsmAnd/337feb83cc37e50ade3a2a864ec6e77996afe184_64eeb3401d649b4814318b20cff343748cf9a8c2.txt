+				(rotation == Surface.ROTATION_90
+						|| rotation == Surface.ROTATION_270) && width > height) {
+			switch(rotation) {
+				case Surface.ROTATION_0:
+					orientation = ActivityInfo.SCREEN_ORIENTATION_PORTRAIT;
