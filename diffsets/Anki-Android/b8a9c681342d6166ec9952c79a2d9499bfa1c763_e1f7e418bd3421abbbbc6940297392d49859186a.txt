+            setNextTime(nextTime)
+        }
+
+        fun setNextTime(nextTime: String?) {
+            text =
+                if (nextTime != null) {
+                    buildSpannedString {
+                        inSpans(RelativeSizeSpan(0.8F)) {
+                            append(nextTime)
+                        }
+                        append("\n")
+                        append(easeName)
+                    }
+                } else {
+                    easeName
+                }
+        }
+    }
