+fun checkRequestPermissionState(
+    activity: Activity
+): RequestPermissionState {
+    return if (RequestStoragePermissions.isPartialGrant(activity)) {

+    } else if (RequestStoragePermissions.checkSelfPermission(activity)) {
+        PermissionGranted
+    } else {
+        PermissionDenied(
