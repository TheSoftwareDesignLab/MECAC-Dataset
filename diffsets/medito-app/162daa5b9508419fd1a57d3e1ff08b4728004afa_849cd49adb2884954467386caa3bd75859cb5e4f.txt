+    var totalScrollableArea = maxScrollExtent;
+
+    scrollProportion = 1 - (scrollPosition / totalScrollableArea);

+    scrollProportion = Curves.easeInOutCubic.transform(scrollProportion);

+    notifyListeners();
+  }
