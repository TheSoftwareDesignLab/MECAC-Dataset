+    private func calculateSize(_ proposal: ProposedViewSize) -> CGSize {
+      return switch (proposal.width, proposal.height) {

+        CGSize(width: originalWidth, height: min(height, originalWidth))
+      case (0, _):
+        CGSize.zero
+      case let (.some(width), _):
