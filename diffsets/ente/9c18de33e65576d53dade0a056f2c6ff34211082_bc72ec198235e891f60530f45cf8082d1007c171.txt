+        let entry = FileEntry(date: Date(), index: nil, base64: nil)

+    // Function to convert image file at URL to base64 string
+    func base64StringFromImage(at fileURL: URL) -> String? {
+        do {
+            let imageData = try Data(contentsOf: fileURL)  // Load image data
+            let base64String = imageData.base64EncodedString(options: .lineLength64Characters)  // Convert to Base64
+            return base64String
+        } catch {
+            print("Error reading image file: \(error.localizedDescription)")
+            return nil
+        }
+    }
+
+    func getTimeline(in context: Context, completion: @escaping (Timeline<Entry>) -> Void) {
+        var entries: [FileEntry] = []
+
+        // Generate a timeline consisting of five entries an hour apart, starting from the current date.
+        let currentDate = Calendar.current.nextDate(
+            after: Date(), matching: DateComponents(second: 0), matchingPolicy: .nextTime,
+            direction: .backward)!
+
+        let totalSet =
+            data?.integer(forKey: "totalMemories")
+        let prefixPath =
+            data?.string(forKey: "memoryPrefix")
+
+        if totalSet != nil && prefixPath != nil {
+            for offset in 0..<(totalSet! / 2) {
+                let randomInt = Int.random(in: 0..<totalSet!)
+                let entryDate = Calendar.current.date(
+                    byAdding: .minute, value: X * offset, to: currentDate)!
+                let base64String = base64StringFromImage(
+                    at: URL(fileURLWithPath: prefixPath! + "/\(randomInt).jpg"))!
+                let entry = FileEntry(date: entryDate, index: randomInt, base64: base64String)
+                entries.append(entry)
+            }
+        } else {
+            let entry = FileEntry(date: Date(), index: nil, base64: nil)
+            entries.append(entry)
+        }
+
+        let timeline = Timeline(entries: entries, policy: .atEnd)
+        completion(timeline)
+    }
